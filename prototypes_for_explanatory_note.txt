ПРИЛОЖЕНИЕ. ПРОТОТИПЫ ОСНОВНЫХ ФУНКЦИЙ И МЕТОДОВ ПРОГРАММЫ

Проект представляет собой серверную часть веб-приложения для игры в Blackjack. Приложение реализовано на TypeScript/Node.js с использованием Express, Socket.IO, Redis и PostgreSQL. Ниже приведены прототипы основных функций, методов и обработчиков без тел реализаций. Комментарии поясняют назначение каждого элемента программы.

/**
 * Состояние одной руки игрока, передаваемое клиенту.
 */
interface HandState {
    bet: number;
    cards: string[];
    status: HandStatus;
    score: number;
}


/**
 * Состояние игрока в игровой комнате.
 */
interface PlayerState {
    userId: number;
    name: string;
    balance: number;
    currentHandIndex: number;
    hands: HandState[];
}


/**
 * Полное состояние игровой комнаты.
 */
interface RoomState {
    roomId: string;
    status: RoomStatus;
    currentPlayerId: number | null;
    dealer: {
        cards: string[];
        score: number;
    };
    players: PlayerState[];
    deckRemaining: number;
}


/**
 * Событие игрового действия игрока.
 */
interface PlayerActionEvent {
    userId: number;
    action: "HIT" | "STAND" | "DOUBLE" | "SPLIT" | "SURRENDER";
    hands: HandState[];
    currentHandIndex: number;
    balance?: number;
}


/**
 * Результат раунда для одного игрока.
 */
interface RoundResultEntry {
    userId: number;
    name: string;
    result: RoundResult;
    payout: number;
    netProfit: number;
    newBalance: number;
    hands: HandState[];
}


/**
 * Итоговое событие завершения раунда.
 */
interface RoundResultEvent {
    results: RoundResultEntry[];
    dealer: {
        cards: string[];
        score: number;
    };
}


/**
 * Описывает события, которые клиент может отправлять серверу.
 */
interface ClientToServerEvents;


/**
 * Описывает события, которые сервер может отправлять клиенту.
 */
interface ServerToClientEvents;


/**
 * Дополнительные данные, хранящиеся внутри socket.data.
 */
interface SocketData {
    userSession: UserSession | null;
    sessionToken: string | null;
    currentRoomId: string | null;
}
